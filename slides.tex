\begin{frame}{Project Overview}
\textbf{What:} A Telegram chatbot that helps students with campus life.

\textbf{Bot:} \texttt{@comp7940\_campus\_bot}

\vspace{0.3cm}
\textbf{Four features:}
\begin{itemize}
\item AI-powered chat (DeepSeek LLM, multi-language)
\item Task and deadline management
\item Campus event discovery with category filtering
\item Study buddy matching based on interests and courses
\end{itemize}

\vspace{0.3cm}
\textbf{Tech stack:} Python, Docker, Redis Cloud, Render, GitHub Actions
\end{frame}

\begin{frame}[fragile]{System Architecture}
\begin{center}
\texttt{User <--> Telegram <--> Bot (Docker/Render)} \\
\texttt{\hspace{4cm} | \hspace{1.5cm} |} \\
\texttt{\hspace{3cm} Redis Cloud \hspace{0.3cm} DeepSeek API}
\end{center}

\vspace{0.3cm}
\begin{itemize}
\item Bot polls Telegram for updates, forwards messages to LLM
\item Conversation history stored in Redis for context
\item All requests logged with latency for monitoring
\item Docker container auto-deployed via Render on git push
\end{itemize}
\end{frame}

\begin{frame}{Security Measures}
\begin{itemize}
\item All secrets stored as environment variables, never in code
\item \texttt{.env} excluded from Git via \texttt{.gitignore}
\item Render stores secrets in encrypted environment variable store
\item Redis connection uses password authentication
